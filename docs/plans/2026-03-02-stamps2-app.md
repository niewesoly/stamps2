            Odkryj wszystkie sprawności ZHR. Przeglądaj, ucz się wymagań i planuj swój szlak.

import { data as routeData } from 'react-router'

import { getBadgeBySlug } from '../../src/data/api'
import StarRating from '../../src/components/StarRating'
import CategoryBadge from '../../src/components/CategoryBadge'

export function meta({ data }: Route.MetaArgs) {
  if (!data) return [{ title: 'Nie znaleziono' }]
  return [
    { title: `${data.badge.name} – ${data.group.spec.name} – Sprawności ZHR` },
    { name: 'description', content: data.badge.requirements[0] ?? '' },
  ]
}

export async function loader({ params }: Route.LoaderArgs) {
  const result = await getBadgeBySlug(params.badgeSlug)
  if (!result) throw routeData(null, { status: 404 })

  const { badge, group } = result
  const prerequisite = badge.basedOn.length > 0
    ? group.spec.badges.find(b => b.id === badge.basedOn[0]) ?? null
    : null

  return { badge, group, prerequisite }
}

## Key Patterns

- **Data loading:** Each route exports `async function loader()` – runs at build-time for SSG
- **SSG prerender:** `react-router.config.ts` exports `prerender()` returning all route paths
- **Slugs:** Polish names → ASCII slugs via `src/data/slugify.ts` (ą→a, ł→l, etc.)
- **Icons:** `https://stamps.zhr.pl/img/form/{uuid.svg}`
- **API:** `https://stamps.zhr.pl/api/badges` – fetched once (cached in module singleton)
